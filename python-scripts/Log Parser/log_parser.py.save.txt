import re
import json
import datetime
from collections import defaultdict

# ── Configuration ────────────────────────────────────────
LOG_FILE = "/home/aj/sample_auth.log"
ALERT_THRESHOLD = 3  # Flag any IP or user with 3+ failed logins
SUSPICIOUS_USERNAMES = ["root", "admin", "administrator", "guest"]

# ── Data Storage ─────────────────────────────────────────
failed_logins = defaultdict(int)
failed_by_ip = defaultdict(int)
failed_by_user = defaultdict(int)
successful_logins = []
alerts = []

print("\n" + "="*55)
print("       AUTOMATED LOG PARSER — SECURITY REPORT")
print("="*55)
print(f"  Log File  : {LOG_FILE}")
print(f"  Scan Time : {datetime.datetime.now()}")
print(f"  Threshold : {ALERT_THRESHOLD}+ failed logins = ALERT")
print("="*55 + "\n")

# ── Parse the Log File ────────────────────────────────────
with open(LOG_FILE, 'r') as f:
    lines = f.readlines()

total_lines = len(lines)
failed_count = 0
success_count = 0

for line in lines:
    line = line.strip()
    if not line:
        continue

    # Extract timestamp, status, user, and IP
    pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) (FAILED LOGIN|SUCCESSFUL LOGIN) user=(\w+) ip=([\d.]+)'
    match = re.search(pattern, line)

    if match:
        timestamp = match.group(1)
        status = match.group(2)
        user = match.group(3)
        ip = match.group(4)

        if status == "FAILED LOGIN":
            failed_count += 1
            failed_by_ip[ip] += 1
            failed_by_user[user] += 1

        elif status == "SUCCESSFUL LOGIN":
            success_count += 1
            successful_logins.append({
                "timestamp": timestamp,
                "user": user,
                "ip": ip
            })

# ── Generate Alerts ───────────────────────────────────────
print("[*] Analyzing for suspicious activity...\n")

# Alert on IPs exceeding threshold
for ip, count in failed_by_ip.items():
    if count >= ALERT_THRESHOLD:
        alert = f"[!] ALERT — IP {ip} had {count} failed logins (possible brute force)"
        alerts.append(alert)
        print(alert)

# Alert on users exceeding threshold
for user, count in failed_by_user.items():
    if count >= ALERT_THRESHOLD:
        alert = f"[!] ALERT — User '{user}' had {count} failed login attempts"
        alerts.append(alert)
        print(alert)

# Alert on suspicious usernames being targeted
for user in failed_by_user:
    if user.lower() in SUSPICIOUS_USERNAMES:
        alert = f"[!] ALERT — Sensitive account targeted: '{user}'"
        if alert not in alerts:
            alerts.append(alert)
            print(alert)

# ── Print Summary ─────────────────────────────────────────
print("\n" + "="*55)
print("                    SUMMARY")
print("="*55)
print(f"  Total log lines parsed : {total_lines}")
print(f"  Failed login attempts  : {failed_count}")
print(f"  Successful logins      : {success_count}")
print(f"  Total alerts generated : {len(alerts)}")

print("\n  Failed attempts by IP:")
for ip, count in sorted(failed_by_ip.items(),
                        key=lambda x: x[1], reverse=True):
    print(f"    {ip} — {count} attempts")

print("\n  Failed attempts by Username:")
for user, count in sorted(failed_by_user.items(),
                          key=lambda x: x[1], reverse=True):
    print(f"    {user} — {count} attempts")

print("\n  Successful Logins:")
for login in successful_logins:
    print(f"    {login['timestamp']} — "
          f"{login['user']} from {login['ip']}")

# ── Save Report ───────────────────────────────────────────
report = {
    "scan_time": str(datetime.datetime.now()),
    "log_file": LOG_FILE,
    "summary": {
        "total_lines": total_lines,
        "failed_logins": failed_count,
        "successful_logins": success_count,
        "total_alerts": len(alerts)
    },
    "alerts": alerts,
    "failed_by_ip": dict(failed_by_ip),
    "failed_by_user": dict(failed_by_user),
    "successful_logins": successful_logins
}

filename = f"log_report_{datetime.date.today()}.json"
with open(filename, 'w') as f:
    json.dump(report, f, indent=4)

print("\n" + "="*55)
print(f"  [*] Report saved to: {filename}")
print("="*55 + "\n")
